76  void operator()(uint i, uint j)
77  {
78  dp_(i,j).val = -1e100;
79  // rule type E: X -> epsilon
80  if (i==j) {
81  dp_(i,j).update(0.0, Rule::E);
82  } else {
83  // rule type L: X -> a Y
84  if (i+1<=j) {
85  value_type v = dp_(i+1,j).val + q_[i];
86  dp_(i,j).update(v, Rule::L);
87  }
88  // rule type R: X -> Y a
89  if (i<=j-1) {
90  value_type v = dp_(i,j-1).val + q_[j-1];
91  dp_(i,j).update(v, Rule::R);
92  }
93  // rule type P: X -> a Y b
94  if (min_loop_+i+1<=j-1) {
95  value_type v = dp_(i+1,j-1).val + 2*gamma_*bp_(i,j-1);
96  dp_(i,j).update(v, Rule::P);
97  }
98  // rule type B: X -> Y Z
99  for (uint k=i+1; k<j; ++k) {
100  value_type v = dp_(i,k).val + dp_(k,j).val;
101  dp_(i,j).update(v, Rule::B, k);
102  }
103  }
104  assert(dp_(i,j).type!=static_cast<Rule::type_t>(-1));
105  }
