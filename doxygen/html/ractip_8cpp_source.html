258 {
259  // centroid_alifoldのコードを流用して、塩基対確率を計算して取り出す関数を作っておいて、呼び出す。

261  //cf->stochastic_fold(seq, num_samples, *out)

263  BPTable bp_centroidfold;
264  bp_centroidfold=cf.get_bp();
265  // bpの変換
266  transBP_centroidfold_ractip(bp_centroidfold, bp, offset, seq);
267 
268  const uint L=seq.size();
269  up.resize(L, VF(1, 1.0));
270  for (uint i=0; i!=L; ++i)
271  {
272  for (uint j=0; j<i; ++j)
273  up[i][0] -= bp[offset[j+1]+(i+1)];
274  for (uint j=i+1; j<L; ++j)
275  up[i][0] -= bp[offset[i+1]+(j+1)];
276  up[i][0] = std::max(0.0f, up[i][0]);
277  }

525 void
526 RactIP::
527 load_from_rip(const char* filename,
528  const std::string& s1, const std::string& s2,
529  VF& bp1, VI& offset1, VF& bp2, VI& offset2, VVF& hp) const
530 {
531  enum { NONE, TABLE_R, TABLE_S, TABLE_I };
532  uint st=NONE;
533 
534  uint L1=s1.size();
535  bp1.resize((L1+1)*(L1+2)/2);
536  offset1.resize(L1+1);
537  for (uint i=0; i<=L1; ++i)
538  offset1[i] = i*((L1+1)+(L1+1)-i-1)/2;
539 
540  uint L2=s2.size();
541  bp2.resize((L2+1)*(L2+2)/2);
542  offset2.resize(L2+1);
543  for (uint i=0; i<=L2; ++i)
544  offset2[i] = i*((L2+1)+(L2+1)-i-1)/2;
545 
546  hp.resize(L1+1, VF(L2+1));
547 
548  std::ifstream is(filename);
549  std::string l;
550  while (std::getline(is, l))
551  {
552  if (strncmp(l.c_str(), "Table R:", 8)==0) st=TABLE_R;//
553  else if (strncmp(l.c_str(), "Table S:", 8)==0) st=TABLE_S;
554  else if (strncmp(l.c_str(), "Table I:", 8)==0) st=TABLE_I;
555  else if (st!=NONE && l[0]>='0' && l[0]<='9')
556  {
557  int i,j;
558  double p;
559  std::istringstream s(l.c_str());
560  s >> i >> j >> p;
561  switch (st)
562  {
563  case TABLE_R://reading
564  bp1[offset1[i]+j] = p;
565  break;
566  case TABLE_S:
567  bp2[offset2[L2-j+1]+L2-i+1] = p;
568  break;
569  case TABLE_I:
570  hp[i][L2-j+1] = p;
571  break;
572  default:
573  break;
574  }
575  }
576  else st=NONE;
577  }
578 }

939  for (uint j=0; j<i+1; ++j)

1036  //std::cout << "inner for 2" << std::endl;

1038  }
1039  }
1040  }
1041  }
1042
