35 {

79  void operator()(uint i, uint j)
80  {
81  double v=0.0;
82  uint c=0;
83  std::list<BPTablePtr>::const_iterator b = bps_.begin();
84  std::list<std::vector<uint> >::const_iterator idx = idxmap_.begin();
85  while (b!=bps_.end() && idx!=idxmap_.end()) {
86  uint ii=(*idx)[i];
87  uint jj=(*idx)[j];
88  if (ii!=static_cast<uint>(-1) && jj!=static_cast<uint>(-1)) v += (**b)(ii,jj);
89  ++c; ++b; ++idx;
90  }
91  bp_.update(i, j, v/c);
92  }

234 void
235 AveragedModel::
236 compute_expected_accuracy_sampling(const std::string& paren, const Aln& aln,
237  uint num_ea_samples, double& sen, double& ppv, double& mcc)
238 {
239  CYKTable<uint> bp_pos(paren.size(), 0);
240  std::stack<uint> st;
241  for (uint i=0; i<paren.size(); ++i) {
242  if (paren[i]=='(') {
243  st.push(i);
244  } else if (paren[i]==')') {
245  bp_pos(st.top(), i)++;
246  st.pop();
247  }
248  }
249  EncodeBP encoder;
250  BPvecPtr bp = encoder.execute(bp_pos);
251 
252  sen = ppv = mcc = 0.0;
253  int N = 0;
254 
255  std::list<std::string>::const_iterator s;
256  for (s=aln.seq().begin(); s!=aln.seq().end(); ++s)
257  {
258  std::string seq;
259  std::vector<uint> idx;
260  if (paren_.empty())
261  remove_gaps(*s, seq, idx);
262  else
263  cf_->set_constraint(remove_gaps(*s, seq, idx, paren_));
264  std::vector<uint> rev(seq.size());
265  for (uint i=0; i!=idx.size(); ++i)
266  if (idx[i]!=static_cast<uint>(-1)) rev[idx[i]]=i;
267 
268  cf_->prepare_stochastic_traceback(seq);
269  for (int n=0; n!=int(num_ea_samples/aln.num_aln()); ++n) {
270  CYKTable<uint> bp_pos(aln.size(), 0);
271  std::vector<int> paren = cf_->stochastic_traceback(seq);
272  for (uint i=0; i!=paren.size(); ++i) {
273  if (paren[i]>int(i))
274  bp_pos(rev[i-1], rev[paren[i]-1])++;
275  }
276  double TP, TN, FP, FN;
277  get_TP_TN_FP_FN (*encoder.execute(bp_pos), *bp, TP, TN, FP, FN);
278  if (TP+FN!=0) sen += (double) TP / (TP+FN);
279  if (TP+FP!=0) ppv += (double) TP / (TP+FP);
280  if (TP+FP!=0 && TP+FN!=0 && TN+FP!=0 && TN+FN!=0)
281  mcc += (TP*TN-FP*FN)/std::sqrt((TP+FP)*(TP+FN)*(TN+FP)*(TN+FN));
282  ++N;
283  }
284  cf_->clean_stochastic_traceback(seq);
285  }
286 
287  sen /= N;
288  ppv /= N;
289  mcc /= N;
290 }
