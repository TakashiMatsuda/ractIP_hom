320  for (uint i=0; i!=bpv.size(); ++i)

437 template <class SEQ>
438 void
439 FoldingEngine<SEQ>::
440 compute_expected_accuracy(const std::string& paren,
441  double& sen, double& ppv, double& mcc) const
442 {
443  std::vector< std::pair<int, int> > BP;
444  std::vector<int> st;
445  for (uint n=0; n<paren.size(); ++n) {
446  const char c = paren[n];
447  if (c == '(') {
448  st.push_back (n);
449  }
450  else if (c == ')') {
451  int left = st.back();
452  st.pop_back();
453  BP.push_back ( std::make_pair<int, int>(left, n) );
454  }
455  }
456  assert (st.size()==0);
457 
458  int L = paren.size ();
459  int L2 = L*(L-1)/2;
460  int N = BP.size();
461 
462  double sump = 0.0;
463  double etp = 0.0;
464 
465  if (bp_.size() != paren.size())
466  std::cerr << bp_.size() << "!=" << paren.size() << std::endl << paren << std::endl;;
467  assert (bp_.size()==paren.size());
468 
469  for (uint i=0; i<bp_.size(); ++i) {
470  for (uint j = i+1; j<bp_.size(); ++j) {
471  double p = bp_(i,j);
472  sump += p;
473  }
474  }
475 
476  for (std::vector<std::pair<int, int> >::const_iterator it=BP.begin();
477  it != BP.end(); ++it) {
478  etp += bp_(it->first, it->second);
479  }
480 
481  double etn = L2 - N - sump + etp;
482  double efp = N - etp;
483  double efn = sump - etp;
484 
485  sen = ppv = mcc = 0;
486  if (etp+efn!=0) sen = etp / (etp + efn);
487  if (etp+efp!=0) ppv = etp / (etp + efp);
488  if (etp+efp!=0 && etp+efn!=0 && etn+efp!=0 && etn+efn!=0)
489  mcc = (etp*etn-efp*efn) / std::sqrt((etp+efp)*(etp+efn)*(etn+efp)*(etn+efn));
490 }

505 template <class SEQ>
506 void
507 FoldingEngine<SEQ>::
508 compute_expected_accuracy_sampling(const std::string& paren, const SEQ& seq,
509  uint num_ea_samples, double& sen, double& ppv, double& mcc)
510 {
511  CYKTable<uint> bp_pos(paren.size(), 0);
512  std::stack<uint> st;
513  for (uint i=0; i<paren.size(); ++i) {
514  if (paren[i]=='(') {
515  st.push(i);
516  } else if (paren[i]==')') {
517  bp_pos(st.top(), i)++;
518  st.pop();
519  }
520  }
521  EncodeBP encoder;
522  BPvecPtr bp = encoder.execute(bp_pos);
523 
524  sen = ppv = mcc = 0.0;
525  int N = 0;
526 
527  prepare_stochastic_traceback(seq);
528  for (int n=0; n!=int(num_ea_samples); ++n) {
529  CYKTable<uint> bp_pos(seq.size(), 0);
530  std::vector<int> paren = stochastic_traceback(seq);
531  for (uint i=0; i!=paren.size(); ++i) {
532  if (paren[i]>int(i)) {
533  bp_pos(i-1, paren[i]-1)++;
534  }
535  }
536  double TP, TN, FP, FN;
537  get_TP_TN_FP_FN (*encoder.execute(bp_pos), *bp, TP, TN, FP, FN);
538  if (TP+FN!=0) sen += (double) TP / (TP+FN);
539  if (TP+FP!=0) ppv += (double) TP / (TP+FP);
540  if (TP+FP!=0 && TP+FN!=0 && TN+FP!=0 && TN+FN!=0)
541  mcc += (TP*TN-FP*FN)/std::sqrt((TP+FP)*(TP+FN)*(TN+FP)*(TN+FN));
542  ++N;
543  }
544  clean_stochastic_traceback(seq);
545 
546  sen /= N;
547  ppv /= N;
548  mcc /= N;
549 }
