477 static
478 int update_arg(void *field, char **orig_field,
479  unsigned int *field_given, unsigned int *prev_given,
480  char *value, const char *possible_values[],
481  const char *default_value,
482  cmdline_parser_arg_type arg_type,
483  int check_ambiguity, int override,
484  int no_free, int multiple_option,
485  const char *long_opt, char short_opt,
486  const char *additional_error)
487 {
488  char *stop_char = 0;
489  const char *val = value;
490  int found;
491  char **string_field;
492  FIX_UNUSED (field);
493 
494  stop_char = 0;
495  found = 0;
496 
497  if (!multiple_option && prev_given && (*prev_given || (check_ambiguity && *field_given)))
498  {
499  if (short_opt != '-')
500  fprintf (stderr, "%s: `--%s' (`-%c') option given more than once%s\n",
501  package_name, long_opt, short_opt,
502  (additional_error ? additional_error : ""));
503  else
504  fprintf (stderr, "%s: `--%s' option given more than once%s\n",
505  package_name, long_opt,
506  (additional_error ? additional_error : ""));
507  return 1; /* failure */
508  }
509 
510  FIX_UNUSED (default_value);
511 
512  if (field_given && *field_given && ! override)
513  return 0;
514  if (prev_given)
515  (*prev_given)++;
516  if (field_given)
517  (*field_given)++;
518  if (possible_values)
519  val = possible_values[found];
520 
521  switch(arg_type) {
522  case ARG_FLAG:
523  *((int *)field) = !*((int *)field);
524  break;
525  case ARG_INT:
526  if (val) *((int *)field) = strtol (val, &stop_char, 0);
527  break;
528  case ARG_FLOAT:
529  if (val) *((float *)field) = (float)strtod (val, &stop_char);
530  break;
531  case ARG_STRING:
532  if (val) {
533  string_field = (char **)field;
534  if (!no_free && *string_field)
535  free (*string_field); /* free previous string */
536  *string_field = gengetopt_strdup (val);
537  }
538  break;
539  default:
540  break;
541  };
542 
543  /* check numeric conversion */
544  switch(arg_type) {
545  case ARG_INT:
546  case ARG_FLOAT:
547  if (val && !(stop_char && *stop_char == '\0')) {
548  fprintf(stderr, "%s: invalid numeric value: %s\n", package_name, val);
549  return 1; /* failure */
550  }
551  break;
552  default:
553  ;
554  };
555 
556  /* store the original value */
557  switch(arg_type) {
558  case ARG_NO:
559  case ARG_FLAG:
560  break;
561  default:
562  if (value && orig_field) {
563  if (no_free) {
564  *orig_field = value;
565  } else {
566  if (*orig_field)
567  free (*orig_field); /* free previous string */
568  *orig_field = gengetopt_strdup (value);
569  }
570  }
571  };
572 
573  return 0; /* OK */
574 }
