122

124  std::string s;
125  while (std::getline(data, s))
126  {
127  s = Trim(s);
128  if (s.length() == 0) continue;
129 
130  // check for MFA header
131  if (s[0] == '>')
132  {
133  names.push_back(s.substr(1));
134  sequences.push_back("@");
135  }
136 
137  // otherwise process sequence data
138  else

141  for (size_t i = 0; i < s.length(); i++)
142  {
143  if (isspace(s[i])) continue;
144  sequences.back() += s[i];
145  }
146  }
147  }
148 
149  // sanity-checks
150  if (sequences.size() == 0) Error("No sequences read.");
151  if (sequences[0].length() == 1) Error("Zero-length sequence read.");
152  for (size_t i = 1; i < sequences.size(); i++)
153  if (sequences[i].length() != sequences[0].length())
154  Error("Not all sequences have the same length.");
155

157  bool consensus_found = false;
158  size_t i = 0;
159  while (i < sequences.size())
160  {
161  // check for alphabetic characters
162  bool is_consensus = true;
163  for (size_t j = 1; is_consensus && j < sequences[i].length(); j++)
164  if (isalpha(sequences[i][j])) is_consensus = false;
165

167  if (is_consensus)
168  {

170  Error("More than one consensus base-pairing structure found.");
171  else

174  sequences.erase(sequences.begin() + i);
175  names.erase(names.begin() + i);
176  consensus_found = true;
177  continue;
178  }
179  }
180  i++;
181  }
182

184  if (!consensus_found)

187  }
188 }

389 std::vector<int> SStruct::ConvertParensToMapping(const std::string &parens) const
390 {
391  std::vector<int> mapping(parens.length(), UNKNOWN);
392  std::vector<int> stack;
393 
394  Assert(parens[0] == '@', "Invalid parenthesized string.");
395  for (int i = 1; i < int(parens.length()); i++)
396  {
397  switch (parens[i])
398  {
399  case '?': break;
400  case '.': mapping[i] = UNPAIRED; break;
401  case '(': stack.push_back(i); break;
402  case ')':
403  if (stack.size() == 0) Error("Parentheses mismatch.");
404  mapping[i] = stack.back();
405  mapping[stack.back()] = i;
406  stack.pop_back();
407  break;
408  default:
409  Error("Unexpected character '%c' in parenthesized structure.", parens[i]);
410  }
411  }
412  if (stack.size() != 0) Error("Parentheses mismatch.");
413 
414  return mapping;

563 //
564 // Compute average pairwise percent identity for the alignment.
565 // The pairwise PID = # of identities / MIN(len1, len2).
567 
568 double SStruct::ComputePercentIdentity() const
569 {
570  double pid = 0.0;
571  for (size_t i = 0; i < sequences.size(); i++)
572  {
573  for (size_t j = i+1; j < sequences.size(); j++)
574  {
575  int identities = 0;
576  int len1 = 0;
577  int len2 = 0;
578 
579  const std::string &s = sequences[i];
580  const std::string &t = sequences[j];
581 
582  for (size_t k = 0; k < s.length(); k++)
583  {
584  if (isalpha(s[k])) len1++;
585  if (isalpha(t[k])) len2++;
586  if (isalpha(s[k]) && toupper(s[k]) == toupper(t[k])) identities++;
587  }
588 
589  int den = std::min(len1, len2);
590  double pairwise_pid = (den == 0 ? 0.0 : double(identities) / den);
591 
592  pid += pairwise_pid;
593  }
594  }
595 
596  return pid / (sequences.size() * (sequences.size() - 1) / 2);
597 }

600 // SStruct::ComputePositionBasedSequenceWeights()
601 //
602 // Compute sequence weights according to:
603 // Henikoff, S., and Henikoff J. 1994. Position-based
604 // sequence weights. J Mol Biol 243(4):574-578.
606 
607 std::vector<double> SStruct::ComputePositionBasedSequenceWeights() const
608 {
609  std::vector<double> weights(sequences.size(), 0.0);
610  std::vector<int> counts(256);
611 
612  for (size_t i = 1; i < sequences[0].length(); i++)
613  {
614  int diversity = 0;
615  std::fill(counts.begin(), counts.end(), 0);
616 
617  for (size_t j = 0; j < sequences.size(); j++)
618  {

620  ++(counts[BYTE(sequences[j][i])]);
