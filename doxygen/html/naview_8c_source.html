192 static void find_regions(void)
193 /*
194 * Identifies the regions in the structure.
195 */
196 
197 {
198  int i,mate,nb1;
199  logical *mark;
200 
201  nb1 = nbase + 1;
202  mark = (int *) malloc(sizeof(int)*nb1);
203  for (i = 0; i < nb1; i++) mark[i] = false;
204  nregion = 0;
205  for (i=0; i<=nbase; i++) {
206  if ( (mate = bases[i].mate) && !mark[i]) {
207  regions[nregion].start1 = i;
208  regions[nregion].end2 = mate;
209  mark[i] = true;
210  mark[mate] = true;

214  i++,mate--) {
215  mark[i] = mark[mate] = true;
216  bases[i].region = bases[mate].region = &regions[nregion];
217  }

219  regions[nregion].start2 = mate+1;

233 static struct loop *construct_loop(int ibase)
234 /*
235 * Starting at residue ibase, recursively constructs the loop containing
236 * said base and all deeper bases.
237 */
238 
239 {
240  int i,mate;
241  struct loop *retloop,*lp;
242  struct connection *cp;
243  struct region *rp;
244  struct radloop *rlp;
245 
246  retloop = &loops[loop_count++];
247  retloop->nconnection = 0;
248  retloop->connections = (struct connection **) malloc(sizeof(struct connection *));
249  retloop->depth = 0;
250  retloop->number = loop_count;
251  retloop->radius = 0.0;
252  for (rlp = rlphead; rlp; rlp = rlp->next)
253  if (rlp->loopnumber == loop_count) retloop->radius = rlp->radius;
254  i = ibase;
255  do {
256  if ((mate = bases[i].mate) != 0) {
257  rp = bases[i].region;
258  if (!bases[rp->start1].extracted) {
259  if (i == rp->start1) {
260  bases[rp->start1].extracted = true;
261  bases[rp->end1].extracted = true;
262  bases[rp->start2].extracted = true;
263  bases[rp->end2].extracted = true;
264  lp = construct_loop(rp->end1 < nbase ? rp->end1+1 : 0);
265  }
266  else if (i == rp->start2){
267  bases[rp->start2].extracted = true;
268  bases[rp->end2].extracted = true;
269  bases[rp->start1].extracted = true;
270  bases[rp->end1].extracted = true;
271  lp = construct_loop(rp->end2 < nbase ? rp->end2+1 : 0);
272  }
273  else {
274  fprintf(stderr, "naview: Error detected in construct_loop. i = %d not found in region table.\n",i);
275  exit(FATAL_ERROR);
276  }
277  retloop->connections = (struct connection **)
278  realloc(retloop->connections,
279  (++retloop->nconnection+1) *
280  sizeof(struct connection *));
281  retloop->connections[retloop->nconnection-1] = cp =
282  struct_alloc(connection);
283  retloop->connections[retloop->nconnection] = NULL;
284  cp->loop = lp;
285  cp->region = rp;
286  if (i == rp->start1) {
287  cp->start = rp->start1;
288  cp->end = rp->end2;
289  }
290  else {
291  cp->start = rp->start2;
292  cp->end = rp->end1;
293  }
294  cp->extruded = false;
295  cp->broken = false;
296  lp->connections = (struct connection **)
297  realloc(lp->connections,
298  (++lp->nconnection+1) *
299  sizeof(struct connection *));
300  lp->connections[lp->nconnection-1] = cp =
301  struct_alloc(connection);
302  lp->connections[lp->nconnection] = NULL;
303  cp->loop = retloop;
304  cp->region = rp;
305  if (i == rp->start1) {
306  cp->start = rp->start2;
307  cp->end = rp->end1;
308  }
309  else {
310  cp->start = rp->start1;
311  cp->end = rp->end2;
312  }
313  cp->extruded = false;
314  cp->broken = false;
315  }
316  i = mate;
317  }
318  if (++i > nbase) i = 0;
319  }
320  while (i != ibase);
321  return retloop;
322 }

391 static int depth(struct loop *lp)
392 /*
393 * Determines the depth of loop, lp. Depth is defined as the minimum
394 * distance to a leaf loop where a leaf loop is one that has only one
395 * or no connections.
396 */
397 
398 {
399  struct connection *cp,**cpp;
400  int count,ret,d;
401 
402  if (lp->nconnection <= 1) return 0;
403  if (lp->mark) return -1;
404  lp->mark = true;
405  count = 0;
406  ret = 0;
407  for (cpp=lp->connections; (cp = *cpp); cpp++) {
408  d = depth(cp->loop);
409  if (d >= 0) {
410  if (++count == 1) ret = d;
411  else if (ret > d) ret = d;
412  }
413  }
414  lp->mark = false;
415  return ret+1;
416 }

418 static void traverse_loop(struct loop *lp, struct connection *anchor_connection)
419 /*
420 * This is the workhorse of the display program. The algorithm is
421 * recursive based on processing individual loops. Each base pairing
422 * region is displayed using the direction given by the circle diagram,
423 * and the connections between the regions is drawn by equally spaced
424 * points. The radius of the loop is set to minimize the square error
425 * for lengths between sequential bases in the loops. The "correct"
426 * length for base links is 1. If the least squares fitting of the
427 * radius results in loops being less than 1/2 unit apart, then that
428 * segment is extruded.
429 *
430 * The variable, anchor_connection, gives the connection to the loop
431 * processed in an previous level of recursion.
432 */
433 
434 {
435  double xs,ys,xe,ye,xn,yn,angleinc,r;
436  double radius,xc,yc,xo,yo,astart,aend,a;
437  struct connection *cp,*cpnext,**cpp,*acp,*cpprev;
438  int i,j,n,ic;
439  double da,maxang;
440  int count,icstart,icend,icmiddle,icroot;
441  logical done,done_all_connections,rooted;
442  int sign;
443  double midx,midy,nrx,nry,mx,my,vx,vy,dotmv,nmidx,nmidy;
444  int icstart1,icup,icdown,icnext,direction;
445  double dan,dx,dy,rr;
446  double cpx,cpy,cpnextx,cpnexty,cnx,cny,rcn,rc,lnx,lny,rl,ac,acn,sx,sy,dcp;
447  int imaxloop;
448 
449  angleinc = 2 * pi / (nbase+1);
450  acp = NULL;
451  icroot = -1;
452  for (cpp=lp->connections, ic = 0; (cp = *cpp); cpp++, ic++) {
453  /* xs = cos(angleinc*cp->start);
454  ys = sin(angleinc*cp->start);
455  xe = cos(angleinc*cp->end);
456  ye = sin(angleinc*cp->end); */
457  xs = -sin(angleinc*cp->start);
458  ys = cos(angleinc*cp->start);
459  xe = -sin(angleinc*cp->end);
460  ye = cos(angleinc*cp->end);
461  xn = ye-ys;
462  yn = xs-xe;
463  r = sqrt(xn*xn + yn*yn);
464  cp->xrad = xn/r;
465  cp->yrad = yn/r;
466  cp->angle = atan2(yn,xn);
467  if (cp->angle < 0.0) cp->angle += 2*pi;
468  if (anchor_connection != NULL &&
469  anchor_connection->region == cp->region) {
470  acp = cp;
471  icroot = ic;
472  }
473  }
474 
475  set_radius:
476  determine_radius(lp,lencut);
477  radius = lp->radius;
478  if (anchor_connection == NULL) xc = yc = 0.0;
479  else {
480  xo = (bases[acp->start].x+bases[acp->end].x) / 2.0;
481  yo = (bases[acp->start].y+bases[acp->end].y) / 2.0;
482  xc = xo - radius * acp->xrad;
483  yc = yo - radius * acp->yrad;
484  }
485 
486  /*
487  * The construction of the connectors will proceed in blocks of
488  * connected connectors, where a connected connector pairs means
489  * two connectors that are forced out of the drawn circle because they
490  * are too close together in angle.
491  */
492 
493  /*
494  * First, find the start of a block of connected connectors
495  */
496 
497  if (icroot == -1)
498  icstart = 0;
499  else icstart = icroot;
500  cp = lp->connections[icstart];
501  count = 0;
502  if (debug) printf("Now processing loop %d\n",lp->number);
503  done = false;
504  do {
505  j = icstart - 1;
506  if (j < 0) j = lp->nconnection - 1;
507  cpprev = lp->connections[j];
508  if (!connected_connection(cpprev,cp)) {
509  done = true;
510  }
511  else {
512  icstart = j;
513  cp = cpprev;
514  }
515  if (++count > lp->nconnection) {
516  /*
517  * Here everything is connected. Break on maximum angular separation
518  * between connections.
519  */
520  maxang = -1.0;
521  for (ic = 0; ic < lp->nconnection; ic++) {
522  j = ic + 1;
523  if (j >= lp->nconnection) j = 0;
524  cp = lp->connections[ic];
525  cpnext = lp->connections[j];
526  ac = cpnext->angle - cp->angle;
527  if (ac < 0.0) ac += 2*pi;
528  if (ac > maxang) {
529  maxang = ac;
530  imaxloop = ic;
531  }
532  }
533  icend = imaxloop;
534  icstart = imaxloop + 1;
535  if (icstart >= lp->nconnection) icstart = 0;
536  cp = lp->connections[icend];
537  cp->broken = true;
538  done = true;
539  }
540  } while (!done);
541  done_all_connections = false;
542  icstart1 = icstart;
543  if (debug) printf("Icstart1 = %d\n",icstart1);
544  while (!done_all_connections) {
545  count = 0;
546  done = false;
547  icend = icstart;
548  rooted = false;
549  while (!done) {
550  cp = lp->connections[icend];
551  if (icend == icroot) rooted = true;
552  j = icend + 1;
553  if (j >= lp->nconnection) {
554  j = 0;
555  }
556  cpnext = lp->connections[j];
557  if (connected_connection(cp,cpnext)) {
558  if (++count >= lp->nconnection) break;
559  icend = j;
560  }
561  else {
562  done = true;
563  }
564  }

566  ic = icup = icdown = icmiddle;
567  if (debug)

569  icstart,icmiddle,icend);

574  ic = icup;
575  }
576  else if (direction == 0) {
577  ic = icmiddle;
578  }
579  else {

581  }

584  if (anchor_connection == NULL || acp != cp) {
585  if (direction == 0) {
586  astart = cp->angle - asin(1.0/2.0/radius);
587  aend = cp->angle + asin(1.0/2.0/radius);

589  bases[cp->start].y = yc + radius*sin(astart);
590  bases[cp->end].x = xc + radius*cos(aend);
591  bases[cp->end].y = yc + radius*sin(aend);
592  }
593  else if (direction < 0) {

595  if (j >= lp->nconnection) j = 0;
596  cp = lp->connections[ic];
597  cpnext = lp->connections[j];
598  cpx = cp->xrad;
599  cpy = cp->yrad;
600  ac = (cp->angle + cpnext->angle) / 2.0;
601  if (cp->angle > cpnext->angle) ac -= pi;
602  cnx = cos(ac);
603  cny = sin(ac);
604  lnx = cny;
605  lny = -cnx;

607  if (da < 0.0) da += 2*pi;
608  if (cp->extruded) {
609  if (da <= pi/2) rl = 2.0;
610  else rl = 1.5;
611  }
612  else {
613  rl = 1.0;
614  }

616  bases[cp->end].y = bases[cpnext->start].y + rl*lny;
617  bases[cp->start].x = bases[cp->end].x + cpy;
618  bases[cp->start].y = bases[cp->end].y - cpx;
619  }

623  cp = lp->connections[j];
624  cpnext = lp->connections[ic];
625  cpnextx = cpnext->xrad;
626  cpnexty = cpnext->yrad;
627  ac = (cp->angle + cpnext->angle) / 2.0;
628  if (cp->angle > cpnext->angle) ac -= pi;
629  cnx = cos(ac);

632  lny = cnx;

676  * that the bisector of the end points is radius distance
677  * away from the loop center.

684  midy = bases[cp->start].y + dy/2.0;

703  dy = bases[cp->start].y - yc;

724  bases[i].y += nmidy - midy;
725  i = cp->end;
726  bases[i].x += nmidx - midx;
727  bases[i].y += nmidy - midy;

739  cpnext = lp->connections[j];

753  if (dcp <= 0.0) dcp += 2*pi;

757  lp->number);
758  }

785  traverse_loop(cp->loop,cp);
786  }
787  }
788  n = 0;
789  sx = 0.0;
790  sy = 0.0;
791  for (ic = 0; ic < lp->nconnection; ic++) {

805  }

818 static void determine_radius(struct loop *lp,double lencut)
819 /*
820 * For the loop pointed to by lp, determine the radius of
821 * the loop that will ensure that each base around the loop will
822 * have a separation of at least lencut around the circle.
823 * If a segment joining two connectors will not support this separation,
824 * then the flag, extruded, will be set in the first of these
825 * two indicators. The radius is set in lp.
826 *
827 * The radius is selected by a least squares procedure where the sum of the
828 * squares of the deviations of length from the ideal value of 1 is used
829 * as the error function.
830 */
831 
832 {
833  double mindit,ci,dt,sumn,sumd,radius,dit;
834  int i,j,end,start,imindit;
835  struct connection *cp,*cpnext;
836  static double rt2_2 = 0.7071068;
837 
838  do {
839  mindit = 1.0e10;
840  for (sumd=0.0, sumn=0.0, i=0;
841  i < lp->nconnection;
842  i++) {
843  cp = lp->connections[i];
844  j = i + 1;
845  if (j >= lp->nconnection) j = 0;
846  cpnext = lp->connections[j];
847  end = cp->end;
848  start = cpnext->start;
849  if (start < end) start += nbase + 1;
850  dt = cpnext->angle - cp->angle;
851  if (dt <= 0.0) dt += 2*pi;
852  if (!cp->extruded)
853  ci = start - end;
854  else {
855  if (dt <= pi/2) ci = 2.0;
856  else ci = 1.5;
857  }
858  sumn += dt * (1.0/ci + 1.0);
859  sumd += dt * dt / ci;
860  dit = dt/ci;
861  if (dit < mindit && !cp->extruded && ci > 1.0) {
862  mindit = dit;
863  imindit = i;
864  }
865  }
866  radius = sumn/sumd;
867  if (radius < rt2_2) radius = rt2_2;
868  if (mindit*radius < lencut) {
869  lp->connections[imindit]->extruded = true;
870  }
871  } while (mindit*radius < lencut);
872  if (lp->radius > 0.0)
873  radius = lp->radius;
874  else lp->radius = radius;
875 }

967 static void construct_circle_segment(int start, int end)
968 /*
969 * Draws the segment of residue between the bases numbered start
970 * through end, where start and end are presumed to be part of a base
971 * pairing region. They are drawn as a circle which has a chord given
972 * by the ends of two base pairing regions defined by the connections.
973 */

975 {
976  double dx,dy,rr,h,angleinc,midx,midy,xn,yn,nrx,nry,mx,my,a;
977  int l,j,i;
978 
979  dx = bases[end].x - bases[start].x;
980  dy = bases[end].y - bases[start].y;
981  rr = sqrt(dx*dx + dy*dy);
982  l = end - start;
983  if (l < 0) l += nbase + 1;
984  if (rr >= l) {
985  dx /= rr;

988  i = start + j;

990  bases[i].x = bases[start].x + dx*(double)j/(double)l;
991  bases[i].y = bases[start].y + dy*(double)j/(double)l;
992  }
993  }
994  else {

999  midy = bases[start].y + dy*rr/2.0;
1000  xn = dy;

1003  nry = midy + h*yn;

1005  my = bases[start].y - nry;

1010  if (i > nbase) i -= nbase + 1;
1011  bases[i].x = nrx + rr*cos(a+j*angleinc);

1022 * connecting circle will fit.

1057  bases[nend].y = bases[end].y - 0.5*dy;

1085  bases[nstart].y = bases[start].y + sin(ac);
1086  start = nstart;

1089  bases[nend].y = bases[end].y + sin(ac);
1090  end = nend;

1092  }

1097 static void find_center_for_arc(int n,double b,double *hp,double *thetap)
1098 /*
1099 * Given n points to be placed equidistantly and equiangularly on a
1100 * polygon which has a chord of length, b, find the distance, h, from the
1101 * midpoint of the chord for the center of polygon. Positive values
1102 * mean the center is within the polygon and the chord, whereas
1103 * negative values mean the center is outside the chord. Also, the
1104 * radial angle for each polygon side is returned in theta.
1105 *
1106 * The procedure uses a bisection algorithm to find the correct
1107 * value for the center. Two equations are solved, the angles
1108 * around the center must add to 2*pi, and the sides of the polygon
1109 * excluding the chord must have a length of 1.
1110 */
1111 
1112 {
1113  double h,hhi,hlow,r,disc,theta,e,phi;
1114  int iter;
1115 #define maxiter 500
1116 
1117  hhi = (n+1) / pi;
1118  hlow = - hhi - b/(n+1.000001-b); /* changed to prevent div by zero if (ih) */
1119  if (b<1) hlow = 0; /* otherwise we might fail below (ih) */
1120  iter = 0;
1121  do {
1122  h = (hhi + hlow) / 2.0;
1123  r = sqrt(h*h + b*b/4.0);
1124  /* if (r<0.5) {r = 0.5; h = 0.5*sqrt(1-b*b);} */
1125  disc = 1.0 - 0.5/(r*r);
1126  if (fabs(disc) > 1.0) {
1127  fprintf(stderr, "Unexpected large magnitude discriminant = %g %g\n", disc,r);
1128  exit(FATAL_ERROR);
1129  }
1130  theta = acos(disc);
1131  /* theta = 2*acos(sqrt(1-1/(4*r*r))); */
1132  phi = acos(h/r);
1133  e = theta * (n+1) + 2*phi - 2*pi;
1134  if (e > 0.0) {
1135  hlow = h;
1136  }
1137  else {
1138  hhi = h;
1139  }
1140  } while (fabs(e) > 0.0001 && ++iter < maxiter);
1141  if (iter >= maxiter) {
1142  fprintf(stderr, "Iteration failed in find_center_for_arc\n");
1143  h = 0.0;
1144  theta = 0.0;
1145  }
1146  *hp = h;
1147  *thetap = theta;
1148 }
