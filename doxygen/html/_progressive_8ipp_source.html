176 template<class RealT>
177 std::string Progressive<RealT>::ComputeAlignment(const RealT *scoring, int LX, int LY)
178 {
179  const int SIZE = (LX+1)*(LY+1);
180  RealT *best = new RealT[SIZE];
181  int *traceback = new int[SIZE];
182 
183  // dynamic programming
184 
185  for (int i = 0; i <= LX; i++)
186  {
187  for (int j = 0; j <= LY; j++)
188  {
189  int ij = i*(LY+1)+j;
190  int i1j = ij-LY-1;
191  int ij1 = ij-1;
192  int i1j1 = ij-LY-2;
193 
194  best[ij] = (ij == 0 ? RealT(0) : RealT(NEG_INF));
195  traceback[ij] = -1;
196 
197  if (i > 0 && j > 0 && best[ij] < best[i1j1] + scoring[ij])
198  {
199  best[ij] = best[i1j1] + scoring[ij];
200  traceback[ij] = 0;
201  }
202 
203  if (i > 0 && best[ij] < best[i1j])
204  {
205  best[ij] = best[i1j];
206  traceback[ij] = 1;
207  }
208 
209  if (j > 0 && best[ij] < best[ij1])
210  {
211  best[ij] = best[ij1];
212  traceback[ij] = 2;
213  }
214  }
215  }
216 
217  delete [] best;
218 
219  // traceback
220 
221  std::string edit_string;
222  int ci = LX;
223  int cj = LY;
224 
225  while (ci > 0 || cj > 0)
226  {
227  switch (traceback[ci*(LY+1)+cj])
228  {
229  case 0: edit_string.push_back('M'); ci--; cj--; break;
230  case 1: edit_string.push_back('X'); ci--; break;
231  case 2: edit_string.push_back('Y'); cj--; break;
232  default: Error("Should not get here."); break;
233  }
234  }
235

237

239  std::reverse(edit_string.begin(), edit_string.end());
240 
241  return edit_string;
242 }
