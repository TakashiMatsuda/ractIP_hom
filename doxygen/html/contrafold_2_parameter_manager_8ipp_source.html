166 void ParameterManager<RealT>::ReadFromFile(const std::string &filename, std::vector<RealT> &values)
167 {
168  std::map<std::string, RealT> params;
169  RealT value;
170  std::string name;
171  std::string s;
172 
173  // read parameter file
174  std::ifstream infile(filename.c_str());
175  if (infile.fail()) Error(("Could not open file \"" + filename + "\" for reading.").c_str());
176 
177  while (getline(infile, s))
178  {
179  // skip blank lines and comments
180  if (s.length() == 0 || s[0] == '#') continue;
181 
182  // read parameter names and values
183  std::istringstream iss(s);
184  if (iss >> name >> value)
185  {
186  if (params.find(name) != params.end())
187  Error("Parameter file contains a duplicate parameter: %s", name.c_str());
188  params[name] = value;
189  }
190  }
191  infile.close();
192 
193  // convert read parameters to vector format
194  values.clear();
195  values.resize(names.size());
196  for (size_t i = 0; i < names.size(); i++)
197  {
198  typename std::map<std::string, RealT>::iterator iter = params.find(names[i]);
199  if (iter == params.end()) Error("Parameter file missing parameter: %s", names[i].c_str());
200  values[i] = iter->second;
201  params.erase(iter);
202  }
203 
204  // print an error message for extra parameters
205  for (typename std::map<std::string, RealT>::iterator iter = params.begin(); iter != params.end(); ++iter)
206  Warning("Parameter file contains extra parameter: %s", iter->first.c_str());
207 }
