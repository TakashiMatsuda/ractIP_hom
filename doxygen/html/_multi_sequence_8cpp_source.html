590 // MultiSequence::ComputePositionBasedSequenceWeights()
591 //
592 // Compute sequence weights according to:
593 // Henikoff, S., and Henikoff J. 1994. Position-based
594 // sequence weights. J Mol Biol 243(4):574-578.
596 
597 std::vector<double> MultiSequence::ComputePositionBasedSequenceWeights() const
598 {
599  std::vector<double> weights(sequences.size(), 0.0);
600  std::vector<int> counts(256);
601 
602  for (int i = 1; i <= sequences[0]->GetLength(); i++)
603  {
604  int diversity = 0;
605  std::fill(counts.begin(), counts.end(), 0);
606 
607  for (size_t j = 0; j < sequences.size(); j++)
608  {
609  if (counts[BYTE(sequences[j]->GetData()[i])] == 0) diversity++;
610  ++(counts[BYTE(sequences[j]->GetData()[i])]);
611  }
612 
613  for (size_t j = 0; j < sequences.size(); j++)
614  weights[j] += 1.0 / (diversity * counts[BYTE(sequences[j]->GetData()[i])]);
615  }
616 
617  weights /= Sum(weights);
618  return weights;
619 }
