70 LogAdd(double x, double y)
71 {
72  if (x<=log(0.0)) return y;
73  if (y<=log(0.0)) return x;
74  return x>y ? log1p(exp(y-x))+x : log1p(exp(x-y))+y;
75 }

78 pf_duplex(const char *s1, const char *s2)
79 {
80  int i, j;
81  double Esum;
82  double kT = (temperature+K0)*GASCONST;
83  n1 = (int) strlen(s1);
84  n2 = (int) strlen(s2);
85 
86  if ((!P) || (fabs(P->temperature - temperature)>1e-6)) {
87  update_fold_params(); P = scale_parameters();
88  make_pair_matrix();
89  }
90 
91  encode_seqs(s1, s2);
92 
93  Esum = pf_duplex_fw();
94  pf_duplex_bk();
95 
96  pr_duplex = (double **) space(sizeof(double *) * (n1+1));
97  for (i=1; i<=n1; i++) pr_duplex[i] = (double *) space(sizeof(double) * (n2+1));
98  pr_duplex2 = (double **) space(sizeof(double *) * (n1+1));
99  for (i=1; i<=n1; i++) pr_duplex2[i] = (double *) space(sizeof(double) * (n2+1));
100 
101  for (i=1; i<=n1; i++) {
102  for (j=n2; j>0; j--) {
103  int type, type2, E;
104  type = pair[S1[i]][S2[j]];
105  pr_duplex[i][j] = exp(fw[i][j]+bk[i][j]-Esum);
106  if (i>1 && j<n2) {
107  type2 = pair[S1[i-1]][S2[j+1]];
108  if (type2) {
109 #ifdef HAVE_VIENNA20
110  E = E_IntLoop(0, 0, type2, rtype[type],
111  SS1[i], SS2[j], SS1[i-1], SS2[j+1], P);
112 #else
113  E = LoopEnergy(0, 0, type2, rtype[type],
114  SS1[i], SS2[j], SS1[i-1], SS2[j+1]);
115 #endif
116  pr_duplex2[i][j] = exp(fw[i-1][j+1]+bk[i][j]-E*10./kT-Esum);
117  }
118  }
119  }
120  }
121 
122  for (i=1; i<=n1; i++) {
123  free(fw[i]);
124  free(bk[i]);
125  }
126  free(fw);
127  free(bk);
128 
129  return Esum;
130 }

142 PRIVATE
143 double
144 pf_duplex_fw()
145 {
146  int i, j;
147  double Esum=log(0.0);
148  double kT = (temperature+K0)*GASCONST;
149 
150  fw = (double **) space(sizeof(double *) * (n1+1));
151  for (i=1; i<=n1; i++) fw[i] = (double *) space(sizeof(double) * (n2+1));
152 
153  for (i=1; i<=n1; i++) {
154  for (j=n2; j>0; j--) {
155  int type, type2, E, k,l;
156  type = pair[S1[i]][S2[j]];
157  fw[i][j] = log(0.0);
158  if (!type) continue;
159  E = P->DuplexInit;
160  if (i>1) E += P->dangle5[type][SS1[i-1]];
161  if (j<n2) E += P->dangle3[type][SS2[j+1]];
162  if (type>2) E += P->TerminalAU;
163  fw[i][j] = -E*10./kT;
164  for (k=i-1; k>0 && k>i-MAXLOOP-2; k--) {
165  for (l=j+1; l<=n2; l++) {
166  if (i-k+l-j-2>MAXLOOP) break;
167  type2 = pair[S1[k]][S2[l]];
168  if (!type2) continue;
169 #ifdef HAVE_VIENNA20
170  E = E_IntLoop(i-k-1, l-j-1, type2, rtype[type],
171  SS1[k+1], SS2[l-1], SS1[i-1], SS2[j+1], P);
172 #else
173  E = LoopEnergy(i-k-1, l-j-1, type2, rtype[type],
174  SS1[k+1], SS2[l-1], SS1[i-1], SS2[j+1]);
175 #endif
176  fw[i][j] = LogAdd(fw[i][j], fw[k][l]-E*10./kT);
177  }
178  }
179  E = 0.;
180  if (i<n1) E += P->dangle3[rtype[type]][SS1[i+1]];
181  if (j>1) E += P->dangle5[rtype[type]][SS2[j-1]];
182  if (type>2) E += P->TerminalAU;
183  Esum = LogAdd(Esum, fw[i][j]-E*10./kT);
184  }
185  }
186 
187  return Esum;
188 }

242 PRIVATE short * encode_seq(const char *sequence) {
243  unsigned int i,l;
244  short *S;
245  l = strlen(sequence);
246  S = (short *) space(sizeof(short)*(l+2));
247  S[0] = (short) l;
248 
249  /* make numerical encoding of sequence */
250  for (i=1; i<=l; i++)
251  S[i]= (short) encode_char(toupper(sequence[i-1]));
252 
253  /* for circular folding add first base at position n+1 */
254  S[l+1] = S[1];
255 
256  return S;
