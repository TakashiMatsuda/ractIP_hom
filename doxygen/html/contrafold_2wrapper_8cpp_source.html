383  for (unsigned int i=0; i!=en_.size(); ++i) if (en_[i]) delete en_[i];
384  en_.resize(aln.size(), NULL);
385  idx_.clear(); idx_.resize(aln.size());
386  rev_.clear(); rev_.resize(aln.size());
387  for (unsigned int i=0; i!=aln.size(); ++i)
388  {
389  std::string seq;
390  idx_[i].resize(aln[i].size()+1);
391  idx_[i][0] = -1u;
392  rev_[i].resize(aln[i].size()+1);
393  for (unsigned int j=0, k=0; j!=aln[i].size(); ++j)
394  {
395  if (aln[i][j]=='-')
396  {
397  idx_[i][j+1] = -1u;
398  }
399  else
400  {
401  idx_[i][j+1] = k+1;
402  rev_[i][k+1] = j+1;
403  seq.push_back(aln[i][j]);
404  k++;
405  }
406  }
407 
408  SStruct s("unknown", seq);
409  en_[i] = new InferenceEngine<T>(!canonical_only_, max_bp_dist_);
410  en_[i]->RegisterParameters(pm_);

412  en_[i]->LoadValues(w_);
