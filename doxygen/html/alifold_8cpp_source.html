114 void
115 AliFoldModel::
116 calculate_posterior(const Aln& aln)
117 {
118  bp_.resize(aln.size());
119  char** seqs=alloc_aln(aln);
120  char* str2=new char[aln.size()+1];
121  int bk=Vienna::fold_constrained;
122 
123  if (!str_.empty())
124  {
125  Vienna::fold_constrained=1;
126  strcpy(str2, str_.c_str());
127  }
128  // scaling parameters to avoid overflow
129 #ifdef HAVE_VIENNA20
130  double min_en = Vienna::alifold((const char**)seqs, str2);
131 #else
132  double min_en = Vienna::alifold(seqs, str2);
133 #endif
134  double kT = (Vienna::temperature+273.15)*1.98717/1000.; /* in Kcal */
135  Vienna::pf_scale = exp(-(1.07*min_en)/kT/aln.size());
136  Vienna::free_alifold_arrays();
137 
138 #ifdef HAVE_VIENNA18
139  Vienna::plist* pi;
140 #else
141  Vienna::pair_info* pi;
142 #endif
143  if (!str_.empty()) strcpy(str2, str_.c_str());
144 #ifdef HAVE_VIENNA20
145  Vienna::alipf_fold((const char**)seqs, str2, &pi);
146 #else
147  Vienna::alipf_fold(seqs, str2, &pi);
148 #endif
149  for (uint k=0; pi[k].i!=0; ++k)
150  bp_.update(pi[k].i-1, pi[k].j-1, pi[k].p);
151  free(pi);
152 
153  Vienna::free_alipf_arrays();
154  if (str2) delete[] str2;
155  free_aln(seqs);
156  Vienna::fold_constrained = bk;
157 }
