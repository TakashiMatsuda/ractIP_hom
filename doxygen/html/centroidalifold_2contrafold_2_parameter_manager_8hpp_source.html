1 // ParameterManager.hpp
3 //
4 // This class is used for parameter management for models in which
5 //
6 // (1) the number of logical (free) parameters in the model is not
7 // the same as the number of physical parameters stored in
8 // memory
9 //
10 // (2) certain sets of logical parameters are bound together in
11 // parameter groups.
12 //
13 // To accomplish this, the ParameterManager class provides a set of
14 // routines for constructing a mapping between logical parameters and
15 // physical parameters:
16 //
17 // AddParameterGroup(group_name)
18 // AddParameterMapping(logical_name, physical_ptr)
19 //
20 // -------------------------------------------------------------------
21 //
22 // The ParameterManager class provides the following functionality:
23 //
24 // GetPhysicalParameters(logical_index)
25 //
26 // return a vector containing pointers to all physical
27 // parameters which map to a particular logical_index (or all
28 // physical parameters if logical_index == -1)
29 //
30 // GetLogicalIndex(physical_ptr)
31 //
32 // return the index of the logical parameter corresponding to
33 // physical_ptr
34 //
35 // GetNumPhysicalParameters()
36 //
37 // return the number of physical parameters
38 //
39 // GetNumLogicalParameters()
40 //
41 // return the number of logical parameters
42 //
