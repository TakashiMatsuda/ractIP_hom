121 static void hinsert(int i_sequence) {
122  int code = hcode(i_sequence);
123  hentry *e, *e2 = &entries[i_sequence];
124 
125  for (e = htable[code]; e; e = e->next)
126  if (strncmp(&s_[e->i_sequence], &s_[i_sequence], k_ - 1) == 0) {
127  e2->i_sequence = e->i_sequence;
128  e2->i_vertices = e->i_vertices;
129  return;
130  }
131  e2->i_sequence = i_sequence;
132  e2->i_vertices = n_vertices++;
133  e2->next = htable[code];
134  htable[code] = e2;
135 }

139 void shuffle1(const char *s, int l, int k) {
140  int i, j, n_lets;
141 
142  s_ = s;
143  l_ = l;
144  k_ = k;
145  if (k_ >= l_ || k_ <= 1) /* two special cases */
146  return;
147 
148  /* use hashtable to find distinct vertices */
149  n_lets = l_ - k_ + 2; /* number of (k-1)-lets */
150  n_vertices = 0;
151  hinit(n_lets);
152  for (i = 0; i < n_lets; i++)
153  hinsert(i);
154  root = entries[n_lets - 1].i_vertices; /* the last let */
155  if (vertices)
156  free(vertices);
157  vertices = malloc0(n_vertices * sizeof(vertex));
158 
159  /* set i_sequence and n_indices for each vertex */
160  for (i = 0; i < n_lets; i++) { /* for each let */
161  hentry *ev = &entries[i];
162  vertex *v = &vertices[ev->i_vertices];
163 
164  v->i_sequence = ev->i_sequence;
165  if (i < n_lets - 1) /* not the last let */
166  v->n_indices++;
167  }
168 
169  /* distribute indices for each vertex */
170  if (indices)
171  free(indices);
172  indices = malloc0((n_lets - 1) * sizeof(int));
173  j = 0;
174  for (i = 0; i < n_vertices; i++) { /* for each vertex */
175  vertex *v = &vertices[i];
176 
177  v->indices = indices + j;
178  j += v->n_indices;
179  }
180 
181  /* populate indices for each vertex */
182  for (i = 0; i < n_lets - 1; i++) { /* for each edge */
183  hentry *eu = &entries[i];
184  hentry *ev = &entries[i + 1];
185  vertex *u = &vertices[eu->i_vertices];
186 
187  u->indices[u->i_indices++] = ev->i_vertices;
188  }
189  hcleanup();
190 }

212 void shuffle2(char *t) {
213  vertex *u, *v;
214  int i, j;
215 
216  /* exact copy case */
217  if (k_ >= l_) {
218  strncpy(t, s_, l_);
219  return;
220  }
221 
222  /* simple permutation case */
223  if (k_ <= 1) {
224  strncpy(t, s_, l_);
225  permutec(t, l_);
226  return;
227  }
228 
229  /* the Wilson algorithm for random arborescence */
230  for (i = 0; i < n_vertices; i++)
231  vertices[i].intree = 0;
232  vertices[root].intree = 1;
233  for (i = 0; i < n_vertices; i++) {
234  u = &vertices[i];
235  while (!u->intree) {
236  u->next = (*randfunc)() % u->n_indices;
237  u = &vertices[u->indices[u->next]];
238  }
239  u = &vertices[i];
240  while (!u->intree) {
241  u->intree = 1;
242  u = &vertices[u->indices[u->next]];
243  }
244  }
245 
246  /* shuffle indices to prepare for walk */
247  for (i = 0; i < n_vertices; i++) {
248  u = &vertices[i];
249  if (i != root) {
250  j = u->indices[u->n_indices - 1]; /* swap the last one */
251  u->indices[u->n_indices - 1] = u->indices[u->next];
252  u->indices[u->next] = j;
253  permutei(u->indices, u->n_indices - 1); /* permute the rest */
254  } else
255  permutei(u->indices, u->n_indices);
256  u->i_indices = 0; /* reset to zero before walk */
257  }
258 
259  /* walk the graph */
260  strncpy(t, s_, k_ - 1); /* the first let remains the same */
261  u = &vertices[0];
262  i = k_ - 1;
263  while (u->i_indices < u->n_indices) {
264  v = &vertices[u->indices[u->i_indices]];
265  j = v->i_sequence + k_ - 2;
266  t[i++] = s_[j];
267  u->i_indices++;
268  u = v;
269  }
270 }
