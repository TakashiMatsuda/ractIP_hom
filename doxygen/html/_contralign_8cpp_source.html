430 template<class RealT>
431 void RunGradientSanityCheck(const Options &options,
432  const std::vector<FileDescription> &descriptions)
433 {
434  // The architecture of the code is somewhat complicated here, so
435  // here's a quick explanation:
436  //
437  // ParameterManager: associates each parameter of the model
438  // with a name and manages hyperparameter
439  // groups
440  //
441  // InferenceEngine: performs application-specific
442  // (loss-augmented) inference
443  //
444  // ComputationEngine: makes all necessary calls to dynamic
445  // programming routines for processing
446  // individual sequences and interfaces with
447  // distributed computation module
448  //
449  // ComputationWrapper: provides a high-level interface for
450  // performing computations on groups of
451  // sequences
452  //
453  // OuterOptimizationWrapper / InnerOptimizationWrapper:
454  // interface between computation routines
455  // and optimization routines
456 
457  ParameterManager<RealT> parameter_manager;
458  InferenceEngine<RealT> inference_engine;
459  inference_engine.RegisterParameters(parameter_manager);
460  ComputationEngine<RealT> computation_engine(options, descriptions, inference_engine, parameter_manager);
461  ComputationWrapper<RealT> computation_wrapper(computation_engine);
462 
463  // decide whether I'm a compute node or master node
464  if (computation_engine.IsComputeNode())
465  {
466  computation_engine.RunAsComputeNode();
467  return;
468  }
469 
470  std::vector<RealT> w(parameter_manager.GetNumLogicalParameters(), RealT(0));
471  computation_wrapper.SanityCheckGradient(computation_wrapper.GetAllUnits(), w);
472  computation_engine.StopComputeNodes();
473 }

489  ComputationWrapper<RealT> computation_wrapper(computation_engine);
490 
491  // decide whether I'm a compute node or master node
492  if (computation_engine.IsComputeNode())
493  {
494  computation_engine.RunAsComputeNode();
495  return;
496  }
497 
498  std::vector<RealT> w(parameter_manager.GetNumLogicalParameters(), RealT(0));
499 
500  for (size_t i = 0; i < w.size(); i++)
501  {
502  w[i] = RealT(drand48() * 1e-4 - 5e-5);

504

506  OptimizationWrapper<RealT> optimization_wrapper(computation_wrapper);

509  // using cross-validation to determine regularization parameters
510  if (options.GetRealValue("holdout_ratio") <= 0)
511  {
512  std::vector<RealT> regularization_coefficients(parameter_manager.GetNumParameterGroups(), options.GetRealValue("regularization_coefficient"));
513  optimization_wrapper.Train(units, w, regularization_coefficients);
514  }
515  else
516  {
517  optimization_wrapper.LearnHyperparameters(units, w);
518  }
519 
520  parameter_manager.WriteToFile("optimize.params.final", w);
521  computation_engine.StopComputeNodes();
